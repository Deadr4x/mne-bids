483# Mapping subject information between MNE-BIDS and MNE-Python. 

484HAND_BIDS_TO_MNE = { 

485 ("n/a",): 0, 

486 ("right", "r", "R", "RIGHT", "Right"): 1, 

487 ("left", "l", "L", "LEFT", "Left"): 2, 

488 ("ambidextrous", "a", "A", "AMBIDEXTROUS", "Ambidextrous"): 3, 

489} 

490 

491HAND_MNE_TO_BIDS = {0: "n/a", 1: "R", 2: "L", 3: "A"} 

492 

493SEX_BIDS_TO_MNE = { 

494 ("n/a", "other", "o", "O", "OTHER", "Other"): 0, 

495 ("male", "m", "M", "MALE", "Male"): 1, 

496 ("female", "f", "F", "FEMALE", "Female"): 2, 

497} 

498 

499SEX_MNE_TO_BIDS = {0: "n/a", 1: "M", 2: "F"} 

500 

501 

502def _map_options(what, key, fro, to): 

503 if what == "sex": 

504 mapping_bids_mne = SEX_BIDS_TO_MNE 

505 mapping_mne_bids = SEX_MNE_TO_BIDS 

506 elif what == "hand": 

507 mapping_bids_mne = HAND_BIDS_TO_MNE 

508 mapping_mne_bids = HAND_MNE_TO_BIDS 

509 else: 

510 raise ValueError("Can only map `sex` and `hand`.") 

511 

512 if fro == "bids" and to == "mne": 

513 # Many-to-one mapping 

514 mapped_option = None 

515 for bids_keys, mne_option in mapping_bids_mne.items(): 

516 if key in bids_keys: 

517 mapped_option = mne_option 

518 break 

519 elif fro == "mne" and to == "bids": 

520 # One-to-one mapping 

521 mapped_option = mapping_mne_bids.get(key, None) 

522 else: 

523 raise RuntimeError( 

524 "fro value {} and to value {} are not " 

525 "accepted. Use 'mne', or 'bids'.".format(fro, to) 

526 ) 

527 

528 return mapped_option
