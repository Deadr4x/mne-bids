63def coil_type(info, idx, ch_type="n/a"): 

64 """Get coil type. 

65 

66 Parameters 

67 ---------- 

68 info : dict 

69 Measurement info 

70 idx : int 

71 Index of channel 

72 ch_type : str 

73 Channel type to fall back upon if a more specific 

74 type is not found 

75 

76 Returns 

77 ------- 

78 type : 'meggradaxial' | 'megrefgradaxial' | 'meggradplanar' 

79 'megmag' | 'megrefmag' | 'eeg' | 'misc' 

80 Type of coil 

81 

82 """ 

83 ch = info["chs"][idx] 

84 for key, values in get_coil_types().items(): 

85 if ch["coil_type"] in values: 

86 return key 

87 return ch_type
