23@verbose 

24def inspect_dataset( 

25 bids_path, 

26 find_flat=True, 

27 l_freq=None, 

28 h_freq=None, 

29 show_annotations=True, 

30 verbose=None, 

31): 

32 """Inspect and annotate BIDS raw data. 

33 

34 This function allows you to browse MEG, EEG, and iEEG raw data stored in a 

35 BIDS dataset. You can toggle the status of a channel (bad or good) by 

36 clicking on the traces, and when closing the browse window, you will be 

37 asked whether you want to save the changes to the existing BIDS dataset or 

38 discard them. 

39 

40 .. warning:: This functionality is still experimental and will be extended 

41 in the future. Its API will likely change. Planned features 

42 include automated bad channel detection and visualization of 

43 MRI images. 

44 

45 .. note:: Currently, only MEG, EEG, and iEEG data can be inspected. 

46 

47 To add or modify annotations, press ``A`` to toggle annotation mode. 

48 

49 Parameters 

50 ---------- 

51 bids_path : BIDSPath 

52 A :class:`mne_bids.BIDSPath` containing at least a ``root``. All 

53 matching files will be inspected. To select only a subset of the data, 

54 set more :class:`mne_bids.BIDSPath` attributes. If ``datatype`` is not 

55 set and multiple datatypes are found, they will be inspected in the 

56 following order: MEG, EEG, iEEG. 

57 To read a specific file, set all the :class:`mne_bids.BIDSPath` 

58 attributes required to uniquely identify the file: If this ``BIDSPath`` 

59 is accepted by :func:`mne_bids.read_raw_bids`, it will work here. 

60 find_flat : bool 

61 Whether to auto-detect channels producing "flat" signals, i.e., with 

62 unusually low variability. Flat **segments** will be added to 

63 ``*_events.tsv``, while channels with more than 5 percent of flat data 

64 will be marked as ``bad`` in ``*_channels.tsv``. 

65 

66 .. note:: 

67 This function calls ``mne.preprocessing.annotate_amplitude`` 

68 (MNE-Python 1.0 or newer) or ``mne.preprocessing.annotate_flat`` 

69 (older versions of MNE-Python) 

70 and will only consider segments of at least **50 ms consecutive 

71 flatness** as "flat" (deviating from MNE-Python's default of 5 ms). 

72 If more than 5 percent of a channel's data has been marked as flat, 

73 the entire channel will be added to the list of bad channels. Only 

74 flat time segments applying to channels **not** marked as bad will 

75 be added to ``*_events.tsv``. 

76 

77 l_freq : float | None 

78 The high-pass filter cutoff frequency to apply when displaying the 

79 data. This can be useful when inspecting data with slow drifts. If 

80 ``None``, no high-pass filter will be applied. 

81 h_freq : float | None 

82 The low-pass filter cutoff frequency to apply when displaying the 

83 data. This can be useful when inspecting data with high-frequency 

84 artifacts. If ``None``, no low-pass filter will be applied. 

85 show_annotations : bool 

86 Whether to show annotations (events, bad segments, …) or not. If 

87 ``False``, toggling annotations mode by pressing ``A`` will be disabled 

88 as well. 

89 %(verbose)s 

90 

91 Examples 

92 -------- 

93 Disable flat channel & segment detection, and apply a filter with a 

94 passband of 1–30 Hz. 

95 

96 >>> from mne_bids import BIDSPath 

97 >>> root = Path('./mne_bids/tests/data/tiny_bids').absolute() 

98 >>> bids_path = BIDSPath(subject='01', task='rest', session='eeg', 

99 ... suffix='eeg', extension='.vhdr', root=root) 

100 >>> inspect_dataset(bids_path=bids_path, find_flat=False, # doctest: +SKIP 

101 ... l_freq=1, h_freq=30) 

102 """ 

103 allowed_extensions = set( 

104 ALLOWED_DATATYPE_EXTENSIONS["meg"] 

105 + ALLOWED_DATATYPE_EXTENSIONS["eeg"] 

106 + ALLOWED_DATATYPE_EXTENSIONS["ieeg"] 

107 ) 

108 

109 bids_paths = [ 

110 p 

111 for p in bids_path.match(check=True) 

112 if (p.extension is None or p.extension in allowed_extensions) 

113 and p.acquisition != "crosstalk" 

114 ] 

115 

116 for bids_path_ in bids_paths: 

117 _inspect_raw( 

118 bids_path=bids_path_, 

119 l_freq=l_freq, 

120 h_freq=h_freq, 

121 find_flat=find_flat, 

122 show_annotations=show_annotations, 

123 )

130def _inspect_raw(*, bids_path, l_freq, h_freq, find_flat, show_annotations): 

131 """Raw data inspection.""" 

132 # Delay the import 

133 import matplotlib 

134 import matplotlib.pyplot as plt 

135 

136 extra_params = dict() 

137 if bids_path.extension == ".fif": 

138 extra_params["allow_maxshield"] = "yes" 

139 raw = read_raw_bids(bids_path, extra_params=extra_params, verbose="error") 

140 old_bads = raw.info["bads"].copy() 

141 old_annotations = raw.annotations.copy() 

142 

143 if find_flat: 

144 raw.load_data() # Speeds up processing dramatically 

145 flat_annot, flat_chans = annotate_amplitude( 

146 raw=raw, flat=0, min_duration=0.05, bad_percent=5 

147 ) 

148 new_annot = raw.annotations + flat_annot 

149 raw.set_annotations(new_annot) 

150 raw.info["bads"] = list(set(raw.info["bads"] + flat_chans)) 

151 del new_annot, flat_annot 

152 else: 

153 flat_chans = [] 

154 

155 show_options = bids_path.datatype == "meg" 

156 

157 with use_browser_backend("matplotlib"): 

158 fig = raw.plot( 

159 title=f"{bids_path.root.name}: {bids_path.basename}", 

160 highpass=l_freq, 

161 lowpass=h_freq, 

162 show_options=show_options, 

163 block=False, 

164 show=False, 

165 verbose="warning", 

166 )

168 # Add our own event handlers so that when the MNE Raw Browser is being 

169 # closed, our dialog box will pop up, asking whether to save changes. 

170 def _handle_close(event): 

171 mne_raw_fig = event.canvas.figure 

172 # Bads alterations are only transferred to `inst` once the figure is 

173 # closed; Annotation changes are immediately reflected in `inst` 

174 new_bads = mne_raw_fig.mne.info["bads"].copy() 

175 new_annotations = mne_raw_fig.mne.inst.annotations.copy() 

176 

177 if not new_annotations: 

178 # Ensure it's not an empty list, but an empty set of Annotations. 

179 new_annotations = mne.Annotations( 

180 onset=[], 

181 duration=[], 

182 description=[], 

183 orig_time=mne_raw_fig.mne.info["meas_date"], 

184 ) 

185 _save_raw_if_changed( 

186 old_bads=old_bads, 

187 new_bads=new_bads, 

188 flat_chans=flat_chans, 

189 old_annotations=old_annotations, 

190 new_annotations=new_annotations, 

191 bids_path=bids_path, 

192 ) 

193 _global_vars["raw_fig"] = None 

194 

195 def _keypress_callback(event): 

196 if event.key == _global_vars["mne_close_key"]: 

197 _handle_close(event) 

198 

199 fig.canvas.mpl_connect("close_event", _handle_close) 

200 fig.canvas.mpl_connect("key_press_event", _keypress_callback) 

201 

202 if not show_annotations: 

203 # Remove annotations and kill `_toggle_annotation_fig` method, since 

204 # we cannot directly and easily remove the associated `a` keyboard 

205 # event callback. 

206 fig._clear_annotations() 

207 fig._toggle_annotation_fig = lambda: None 

208 # Ensure it's not an empty list, but an empty set of Annotations. 

209 old_annotations = mne.Annotations( 

210 onset=[], duration=[], description=[], orig_time=raw.info["meas_date"] 

211 ) 

212 

213 if matplotlib.get_backend() != "agg": 

214 plt.show(block=True) 

215 

216 _global_vars["raw_fig"] = fig 

217 _global_vars["mne_close_key"] = fig.mne.close_key

220def _annotations_almost_equal(old_annotations, new_annotations): 

221 """Allow for a tiny bit of floating point precision loss.""" 

222 if ( 

223 np.array_equal(old_annotations.description, new_annotations.description) 

224 and np.array_equal(old_annotations.orig_time, new_annotations.orig_time) 

225 and np.allclose(old_annotations.onset, new_annotations.onset) 

226 and np.allclose(old_annotations.duration, new_annotations.duration) 

227 ): 

228 return True 

229 else: 

230 return False

233def _save_annotations(*, annotations, bids_path): 

234 # Attach the new Annotations to our raw data so we can easily convert them 

235 # to events, which will be stored in the *_events.tsv sidecar. 

236 extra_params = dict() 

237 if bids_path.extension == ".fif": 

238 extra_params["allow_maxshield"] = "yes" 

239 

240 raw = read_raw_bids( 

241 bids_path=bids_path, extra_params=extra_params, verbose="warning" 

242 ) 

243 raw.set_annotations(annotations) 

244 events, durs, descrs = _read_events( 

245 events=None, event_id=None, bids_path=bids_path, raw=raw 

246 ) 

247 

248 # Write sidecar – or remove it if no events are left. 

249 events_tsv_fname = bids_path.copy().update(suffix="events", extension=".tsv").fpath 

250 

251 if len(events) > 0: 

252 _events_tsv( 

253 events=events, 

254 durations=durs, 

255 raw=raw, 

256 fname=events_tsv_fname, 

257 trial_type=descrs, 

258 overwrite=True, 

259 ) 

260 elif events_tsv_fname.exists(): 

261 logger.info( 

262 f"No events remaining after interactive inspection, " 

263 f"removing {events_tsv_fname.name}" 

264 ) 

265 events_tsv_fname.unlink()

268def _save_raw_if_changed( 

269 *, old_bads, new_bads, flat_chans, old_annotations, new_annotations, bids_path 

270): 

271 """Save bad channel selection if it has been changed. 

272 

273 Parameters 

274 ---------- 

275 old_bads : list 

276 The original bad channels. 

277 new_bads : list 

278 The updated set of bad channels (i.e. **all** of them, not only the 

279 changed ones). 

280 flat_chans : list 

281 The auto-detected flat channels. This is either an empty list or a 

282 subset of ``new_bads``. 

283 old_annotations : mne.Annotations 

284 The original Annotations. 

285 new_annotations : mne.Annotations 

286 The new Annotations. 

287 """ 

288 assert set(flat_chans).issubset(set(new_bads)) 

289 

290 if set(old_bads) == set(new_bads): 

291 bads = None 

292 bad_descriptions = [] 

293 else: 

294 bads = new_bads 

295 bad_descriptions = [] 

296 

297 # Generate entries for the `status_description` column. 

298 channels_tsv_fname = ( 

299 bids_path.copy().update(suffix="channels", extension=".tsv").fpath 

300 ) 

301 channels_tsv_data = _from_tsv(channels_tsv_fname) 

302 

303 for ch_name in bads: 

304 idx = channels_tsv_data["name"].index(ch_name) 

305 if channels_tsv_data["status"][idx] == "bad": 

306 # Channel was already marked as bad in the data, so retain 

307 # existing description. 

308 description = channels_tsv_data["status_description"][idx] 

309 elif ch_name in flat_chans: 

310 description = "Flat channel, auto-detected via MNE-BIDS" 

311 else: 

312 # Channel has been manually marked as bad during inspection 

313 description = "Interactive inspection via MNE-BIDS" 

314 

315 bad_descriptions.append(description) 

316 del ch_name, description 

317 

318 del ( 

319 channels_tsv_data, 

320 channels_tsv_fname, 

321 ) 

322 

323 if _annotations_almost_equal(old_annotations, new_annotations): 

324 annotations = None 

325 else: 

326 annotations = new_annotations 

327 

328 if bads is None and annotations is None: 

329 # Nothing has changed, so we can just exit. 

330 return None 

331 

332 return _save_raw_dialog_box( 

333 bads=bads, 

334 bad_descriptions=bad_descriptions, 

335 annotations=annotations, 

336 bids_path=bids_path, 

337 )

340def _save_raw_dialog_box(*, bads, bad_descriptions, annotations, bids_path): 

341 """Display a dialog box asking whether to save the changes.""" 

342 # Delay the imports 

343 import matplotlib 

344 import matplotlib.pyplot as plt 

345 from matplotlib.widgets import Button 

346 from mne.viz.utils import figure_nobar 

347 

348 title = "Save changes?" 

349 message = "You have modified " 

350 if bads is not None and annotations is None: 

351 message += "the bad channel selection " 

352 figsize = (7.5, 2.5) 

353 elif bads is None and annotations is not None: 

354 message += "the bad segments selection " 

355 figsize = (7.5, 2.5) 

356 else: 

357 message += "the bad channel and\nannotations selection " 

358 figsize = (8.5, 3) 

359 

360 message += ( 

361 f"of\n" 

362 f"{bids_path.basename}.\n\n" 

363 f"Would you like to save these changes to the\n" 

364 f"BIDS dataset?" 

365 ) 

366 icon_fname = str(Path(__file__).parent / "assets" / "help-128px.png") 

367 icon = plt.imread(icon_fname) 

368 

369 fig = figure_nobar(figsize=figsize) 

370 fig.canvas.manager.set_window_title("MNE-BIDS Inspector") 

371 fig.suptitle(title, y=0.95, fontsize="xx-large", fontweight="bold") 

372 

373 gs = fig.add_gridspec(1, 2, width_ratios=(1.5, 5)) 

374 

375 # The dialog box tet. 

376 ax_msg = fig.add_subplot(gs[0, 1]) 

377 ax_msg.text( 

378 x=0, 

379 y=0.8, 

380 s=message, 

381 fontsize="large", 

382 verticalalignment="top", 

383 horizontalalignment="left", 

384 multialignment="left", 

385 ) 

386 ax_msg.axis("off") 

387 

388 # The help icon. 

389 ax_icon = fig.add_subplot(gs[0, 0]) 

390 ax_icon.imshow(icon) 

391 ax_icon.axis("off") 

392 

393 # Buttons. 

394 ax_save = fig.add_axes([0.6, 0.05, 0.3, 0.1]) 

395 ax_dont_save = fig.add_axes([0.1, 0.05, 0.3, 0.1]) 

396 

397 save_button = Button(ax=ax_save, label="Save") 

398 save_button.label.set_fontsize("medium") 

399 save_button.label.set_fontweight("bold") 

400 

401 dont_save_button = Button(ax=ax_dont_save, label="Don't save") 

402 dont_save_button.label.set_fontsize("medium") 

403 dont_save_button.label.set_fontweight("bold") 

404 

405 # Store references to keep buttons alive. 

406 fig.save_button = save_button 

407 fig.dont_save_button = dont_save_button 

408 

409 # Define callback functions. 

410 def _save_callback(event): 

411 plt.close(event.canvas.figure) # Close dialog 

412 _global_vars["dialog_fig"] = None 

413 

414 if bads is not None: 

415 _save_bads(bads=bads, descriptions=bad_descriptions, bids_path=bids_path) 

416 if annotations is not None: 

417 _save_annotations(annotations=annotations, bids_path=bids_path) 

418 

419 def _dont_save_callback(event): 

420 plt.close(event.canvas.figure) # Close dialog 

421 _global_vars["dialog_fig"] = None 

422 

423 def _keypress_callback(event): 

424 if event.key in ["enter", "return"]: 

425 _save_callback(event) 

426 elif event.key == _global_vars["mne_close_key"]: 

427 _dont_save_callback(event) 

428 

429 # Connect events to callback functions. 

430 save_button.on_clicked(_save_callback) 

431 dont_save_button.on_clicked(_dont_save_callback) 

432 fig.canvas.mpl_connect("close_event", _dont_save_callback) 

433 fig.canvas.mpl_connect("key_press_event", _keypress_callback) 

434 

435 if matplotlib.get_backend() != "agg": 

436 fig.show() 

437 

438 _global_vars["dialog_fig"] = fig
